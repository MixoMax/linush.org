# Jupyter notebook converted to Python script.

from bs4 import BeautifulSoup
import matplotlib.pyplot as plt
from tqdm import tqdm
import json

with open("./table_data.html", "r") as file:
    html_content = file.read()

soup = BeautifulSoup(html_content, "html.parser")

country_borders = {}

for row in soup.find_all("tr")[2:]:
    country_th = row.find("th")
    if not country_th:
        continue
    country_name = country_th.get_text(strip=True)

    # Find the cell with the bordering countries
    border_cells = row.find_all("td")
    if len(border_cells) > 1:
        borders_text = border_cells[1].get_text(strip=True)
        # Split the text by '/' and clean up each country name
        bordering_countries = [country.strip() for country in borders_text.split("/")]
        country_borders[country_name] = bordering_countries
    else:
        # If there's no second 'td', it means no bordering countries are listed in that format
        country_borders[country_name] = []


with open("static/country_borders.json", "w") as json_file:
    json.dump(country_borders, json_file, indent=4)

with open("static/country_borders.json", "r") as json_file:
    country_borders = json.load(json_file)


all_connections = set()
for country, borders in country_borders.items():
    for border in borders:
        connection = tuple([country, border])
        connection_reversed = tuple([border, country])
        if connection not in all_connections and connection_reversed not in all_connections:
            all_connections.add(connection)
print(f"Total unique land borders: {len(all_connections)}")
# Output:
#   Total unique land borders: 365


def shortest_path(start_country: str, end_country: str, borders_dict: dict) -> list:
    from collections import deque
    if start_country not in borders_dict or end_country not in borders_dict:
        return []
    if start_country == end_country:
        return [start_country]
    # breadth-first search
    # assign each country a distance to the end country
    # start at the end country with distance 0
    # then assign all bordering countries distance 1
    # then assign all bordering countries of those countries distance 2
    # and so on until we reach the start country
    queue = deque([(end_country, [end_country])])
    visited = set()
    while queue:
        current_country, path = queue.popleft()
        if current_country == start_country:
            return path[::-1]  # Return reversed path
        visited.add(current_country)
        for neighbor in borders_dict.get(current_country, []):
            if neighbor not in visited:
                queue.append((neighbor, path + [neighbor]))
    return []  # No path found

path_lengths = {}
for start_country in tqdm(list(country_borders.keys())):
    for end_country in list(country_borders.keys()):
        if start_country == end_country:
            continue

        if (start_country, end_country) in path_lengths or (end_country, start_country) in path_lengths:
            continue  # Already computed

        path = shortest_path(start_country, end_country, country_borders)
        if path:
            path_lengths[(start_country, end_country)] = len(path) - 1

longest_path = list(sorted(path_lengths.items(), key=lambda x: x[1], reverse=True))[0]
print(f"Longest path is from {longest_path[0][0]} to {longest_path[0][1]} with {longest_path[1]} borders crossed.")
# Output:
#   100%|██████████| 195/195 [00:10<00:00, 17.86it/s]
#   Longest path is from El Salvador to Lesotho with 20 borders crossed.

#   


print(shortest_path("El Salvador", "Lesotho", country_borders))
# Output:
#   ['El Salvador', 'Guatemala', 'Mexico', 'United States', 'Canada', 'Denmark', 'Germany', 'Poland', 'Russia', 'Azerbaijan', 'Turkey', 'Syria', 'Israel', 'Egypt', 'Sudan', 'Central African Republic', 'Democratic Republic of the Congo', 'Zambia', 'Botswana', 'South Africa', 'Lesotho']


length_counts = {}
for length in path_lengths.values():
    if length not in length_counts:
        length_counts[length] = 0
    length_counts[length] += 1
lengths = sorted(length_counts.keys())
counts = [length_counts[length] for length in lengths]
plt.bar(lengths, counts)
# Output:
#   <BarContainer object of 25 artists>
#   <Figure size 640x480 with 1 Axes>

import uvicorn
from fastapi import FastAPI, Response
from fastapi.responses import JSONResponse, FileResponse
from fastapi.middleware.cors import CORSMiddleware
import json
import os
from collections import deque
import requests
import sys

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

with open("static/country_borders.json", "r") as f:
    borders_dict = json.load(f)

@app.get("/api/v1/path")
def shortest_path(start_country: str, end_country: str) -> list:
    global borders_dict
    if start_country not in borders_dict or end_country not in borders_dict:
        return []
    if start_country == end_country:
        return [start_country]
    # breadth-first search
    # assign each country a distance to the end country
    # start at the end country with distance 0
    # then assign all bordering countries distance 1
    # then assign all bordering countries of those countries distance 2
    # and so on until we reach the start country
    queue = deque([(end_country, [end_country])])
    visited = set()
    while queue:
        current_country, path = queue.popleft()
        if current_country == start_country:
            return path[::-1]  # Return reversed path
        visited.add(current_country)
        for neighbor in borders_dict.get(current_country, []):
            if neighbor not in visited:
                queue.append((neighbor, path + [neighbor]))
    return []  # No path found

@app.get("/api/v1/image_proxy")
def image_proxy(url: str):
    forbidden_filesystem_chars = [":", "/", "\\", "?", "%", "*", "|", "\"", "<", ">"]
    clean_file_name = url
    for char in forbidden_filesystem_chars:
        clean_file_name = clean_file_name.replace(char, "_")
        local_path = os.path.join("static", "cached_images", clean_file_name)
    if os.path.exists(local_path):
        return FileResponse(local_path)

    try:
        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'
        }
        
        response = requests.get(url, headers=headers)

        # Cache the image locally
        os.makedirs(os.path.dirname(local_path), exist_ok=True)
        with open(local_path, "wb") as f:
            f.write(response.content)
            f.close()

        return Response(content=response.content, media_type=response.headers['Content-Type'])
    except Exception as e:
        return JSONResponse(content={"error": str(e)}, status_code=500)


@app.get("/{path:path}")
async def get_file(path: str):
    if path == "":
        path = "index.html"
    file_path = os.path.join("static", path)
    if os.path.exists(file_path):
        return FileResponse(file_path)
    else:
        return JSONResponse(content={"error": "File not found"}, status_code=404)
    
if __name__ == "__main__":
    port = 8000
    if len(sys.argv) > 1:
        port = int(sys.argv[1])
    uvicorn.run(app, host="0.0.0.0", port=port)

        const world = Globe()
            (document.getElementById('globeViz'))

            fetch('https://raw.githubusercontent.com/vasturiano/globe.gl/master/example/datasets/ne_110m_admin_0_countries.geojson').then(res => res.json()),

                option2.text = country;

            document.getElementById('findPath').addEventListener('click', () => {

                    .then(path => {

                            const endCountry = findCountryFeature(path[i + 1]);

                                    lng: (bbox.x[0] + bbox.x[1]) / 2,
                                    altitude: calculateAltitude(bbox)
                                }, 1000);
                            }
                        }
                    });
            });
        });

        function findCountryFeature(countryName) {
            // find the feature in countries matching the countryName
            return countries.features.find(feat => {
                const nameCandidates = [
                    feat.properties.ADMIN,
                    feat.properties.BRK_NAME,
                    feat.properties.FORMAL_EN,
                    feat.properties.NAME,
                    feat.properties.NAME_SORT
                ];
                for (let name of nameCandidates) {
                    if (name && name.toLowerCase() === countryName.toLowerCase()) {
                        return true;
                    }
                }
                return false;
            });
        }

        function getCentroid(feature) {
            let largestPolygon = [];
            let maxArea = 0;

            if (feature.geometry.type === 'Polygon') {
                largestPolygon = feature.geometry.coordinates[0];
            } else if (feature.geometry.type === 'MultiPolygon') {
                feature.geometry.coordinates.forEach(polygonCoords => {
                    const polygon = polygonCoords[0];
                    const area = d3.polygonArea(polygon);
                    if (area > maxArea) {
                        maxArea = area;
                        largestPolygon = polygon;
                    }
                });
            }

            if (largestPolygon.length === 0) {
                // Fallback for features with no polygons or empty coordinates
                const bbox = getBoundingBox([feature]);
                return {
                    lat: (bbox.y[0] + bbox.y[1]) / 2,
                    lng: (bbox.x[0] + bbox.x[1]) / 2
                };
            }

            const centroid = d3.polygonCentroid(largestPolygon);
            console.log('Centroid for', feature.properties.ADMIN, centroid);
            console.log(feature.properties);
            return {
                lat: centroid[1],
                lng: centroid[0]
            };
        }

        function getBoundingBox(features) {
            const bbox = { x: [Infinity, -Infinity], y: [Infinity, -Infinity] };
            features.forEach(feature => {
                const coords = feature.geometry.coordinates;
                const geomType = feature.geometry.type;

                const processCoords = (arr) => {
                    arr.forEach(c => {
                        if (typeof c[0] === 'number') {
                            bbox.x[0] = Math.min(bbox.x[0], c[0]);
                            bbox.x[1] = Math.max(bbox.x[1], c[0]);
                            bbox.y[0] = Math.min(bbox.y[0], c[1]);
                            bbox.y[1] = Math.max(bbox.y[1], c[1]);
                        } else {
                            processCoords(c);
                        }
                    });
                };
                processCoords(coords);
            });
            return bbox;
        }

        function calculateAltitude(bbox) {
            const globeWidth = 2; // globe.gl radius is 1, so diameter is 2
            const lngDiff = bbox.x[1] - bbox.x[0];
            const latDiff = bbox.y[1] - bbox.y[0];
            const maxDiff = Math.max(lngDiff, latDiff);
            const zoomFactor = 1.5;
            const altitude = maxDiff / (180 / Math.PI) * zoomFactor;
            return Math.max(altitude, 0.5); // Set a min altitude
        }
