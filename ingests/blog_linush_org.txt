app = FastAPI(title="Blog Server", description="A professional blog server with markdown support")

# Mount static files

# Markdown configuration with extensions
def get_markdown_processor():
    return markdown.Markdown(extensions=[
        'codehilite',
        'toc',
        'tables',
        'fenced_code',
        'footnotes',
        'attr_list',
        'def_list',
        'abbr',
        'nl2br',
        'sane_lists',
        'smarty',
        'wikilinks',
        'meta'
    ], extension_configs={
        'codehilite': {
            'css_class': 'highlight',
            'use_pygments': True,
            'noclasses': True
        },
        'toc': {
            'permalink': True,
            'permalink_class': 'toc-link',
            'permalink_title': 'Link to this section'
        }
    })

def process_markdown_file(file_path: str) -> Dict[str, Any]:
    """Process a markdown file and return metadata and HTML content."""
    if not os.path.exists(file_path):
        raise FileNotFoundError(f"File {file_path} not found")
    
    with open(file_path, 'r', encoding='utf-8') as f:
        post = frontmatter.load(f)
    
    md = get_markdown_processor()
    html_content = md.convert(post.content)
    
    # Extract metadata
    metadata = post.metadata
    
    # Get file stats for date if not in frontmatter
    file_stat = os.stat(file_path)
    created_date = datetime.fromtimestamp(file_stat.st_ctime)
    modified_date = datetime.fromtimestamp(file_stat.st_mtime)
    
    return {
        'title': metadata.get('title', os.path.basename(file_path).replace('.md', '').replace('_', ' ').title()),
        'content': html_content,
        'toc': md.toc if hasattr(md, 'toc') else '',
        'meta': md.Meta if hasattr(md, 'Meta') else {},
        'created_date': metadata.get('date', created_date),
        'modified_date': modified_date,
        'author': metadata.get('author', 'Anonymous'),
        'description': metadata.get('description', ''),
        'tags': metadata.get('tags', []),
        'filename': os.path.basename(file_path).replace('.md', '')
    }

def get_all_blog_posts() -> List[Dict[str, Any]]:
    """Get all blog posts from the blog_src directory."""
    blog_posts = []
    blog_src_dir = Path('blog_src')
    
    if not blog_src_dir.exists():
        return []
    
    for md_file in blog_src_dir.glob('*.md'):
        try:
            post_data = process_markdown_file(str(md_file))
            blog_posts.append(post_data)
        except Exception as e:
            print(f"Error processing {md_file}: {e}")
    
    # Sort by creation date (newest first)
    blog_posts.sort(key=lambda x: x['created_date'], reverse=True)
    return blog_posts

def create_blog_html(post_data: Dict[str, Any]) -> str:
    """Create a complete HTML page for a blog post using a template file."""
    template_path = "templates/blog_template.html"
    with open(template_path, "r", encoding="utf-8") as f:
        template = f.read()

    # Prepare replacements
    replacements = {
        "{{title}}": post_data["title"],
        "{{author}}": post_data["author"],
        "{{created_date}}": post_data["created_date"].strftime("%B %d, %Y"),
        "{{description}}": post_data["description"],
        "{{content}}": post_data["content"],
        "{{toc}}": (
            f'<div class="table-of-contents"><h3>Table of Contents</h3>{post_data["toc"]}</div>'
            if post_data["toc"] else ""
        ),
        "{{tags}}": (
            '<div class="post-tags">' +
            "".join([f'<a href="/tag/{tag}" class="tag">{tag}</a>' for tag in post_data["tags"]]) +
            '</div>' if post_data["tags"] else ""
        ),
    }

    # Replace placeholders
    for key, value in replacements.items():
        template = template.replace(key, value)

    return template

@app.get("/", response_class=HTMLResponse)

@app.get("/blog/{filename}")
async def get_blog_post(filename: str):
    """Get a specific blog post by filename."""
    file_path = f"blog_src/{filename}.md"
    
    try:
        post_data = process_markdown_file(file_path)
        return HTMLResponse(create_blog_html(post_data))
    except FileNotFoundError:
        raise HTTPException(status_code=404, detail="Blog post not found")
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Error processing blog post: {str(e)}")

@app.get("/blogs")

        card += '</div>'
        if post["description"]:
            card += f'<p class="post-description">{post["description"]}</p>'
        if post["tags"]:

            card += "".join([f'<a href="/tag/{tag}" class="tag">{tag}</a>' for tag in post["tags"]])

@app.get("/tag/{tag_name}")
async def get_posts_by_tag(tag_name: str):
    """Get all blog posts that have the specified tag using a template."""
    all_posts = get_all_blog_posts()
    tagged_posts = [post for post in all_posts if tag_name.lower() in [tag.lower() for tag in post['tags']]]

    blog_cards = ""
    if not tagged_posts:
        blog_cards = (
            f'<div class="no-posts">'
            f'<h2>No posts found with tag "{tag_name}"</h2>'
            f'<p><a href="/blogs">View all posts</a> or <a href="/">return to homepage</a></p>'
            f'</div>'
        )
        description = ""
    else:
        for post in tagged_posts:
            card = '<div class="blog-card">'
            card += f'<h2><a href="/blog/{post["filename"]}">{post["title"]}</a></h2>'
            card += '<div class="post-meta">'
            card += f'<span>By {post["author"]}</span>'
            card += f'<span>{post["created_date"].strftime("%B %d, %Y")}</span>'
            card += '</div>'
            if post["description"]:
                card += f'<p class="post-description">{post["description"]}</p>'
            if post["tags"]:
                card += '<div class="post-tags">'
                card += "".join([f'<a href="/tag/{tag}" class="tag">{tag}</a>' for tag in post["tags"]])
                card += '</div>'
            card += '</div>'
            blog_cards += card
        description = f'Showing {len(tagged_posts)} post{"s" if len(tagged_posts) != 1 else ""} with this tag'

    }

    for key, value in replacements.items():
        template = template.replace(key, value)

    return HTMLResponse(template)

# Israel's Nuclear Monopoly in the Middle East

The Middle East is a region fraught with tension and conflict[^1]. While every other region with such a huge conflict potential has multiple actors with nuclear weapons, the Middle East has only one nuclear power: Israel[^2]. This nuclear monopoly creates a situation where Israel, with backing from the U.S., has unchecked power in the region[^3]. Over the past decades, we have seen Israel repeatedly attack its neighbors, including most notably Palestine, Lebanon, and Syria. These attacks have often been justified under the pretext of self-defense or preemptive action against perceived threats[^4]. However, the reality is that these strikes have led to significant loss of life and destabilization of the region. The absence of a balance of power, such as that provided by multiple nuclear states, has made it easier for Israel to act with impunity[^3].

# Mutually Assured Destruction

The concept of Mutually Assured Destruction (MAD) has been a cornerstone of nuclear deterrence theory since the Cold War. The idea is simple: if two opposing sides possess the capability to destroy each other with nuclear weapons, neither side is likely to initiate a conflict.  
While this principle has not stopped opposing nations from making threats or engaging in proxy wars, no nation has ever followed through on a nuclear attack against another nuclear power. The most prominent examples of this are the Russia-Ukraine war and the tensions between North Korea and South Korea.  
In the case of the Russian invasion of Ukraine, Vladimir Putin has repeatedly threatened the use of nuclear weapons, but has never followed through, because if he did, Russia would be destroyed in retaliatory strikes by Ukraine's allies such as the U.S., France and the UK.  
Similarly, North Korea has made numerous threats against South Korea and the U.S., but has never launched a nuclear attack, knowing that such an action would lead to its own destruction by the overwhelming military response from the U.S. and other allies of South Korea.  

In both cases, the presence of nuclear weapons, by the U.S. and its allies, has acted as a deterrent against the use of nuclear weapons for offensive purposes.  
This is also not unique to these two scenarios. The principle of MAD has been a significant factor in preventing nuclear war since the cold war, as it creates a situation where the costs of initiating a nuclear conflict are too high for any rational actor to bear.  


# Iran as a rational actor

From a western perspective, Iran is often portrayed as a rogue state, driven by irrational ideology and a desire to spread its influence across the Middle East.  
However, this view overlooks the fact that Iran is a rational actor, motivated by national interests and security concerns.  
And for Iran, the most significant threat comes from Israel's unbalanced power. The presence of a nuclear-armed Israel, coupled with the lack of any similar capability in Iran, creates a significant security dilemma for Tehran.  

Iran's desire to develop nuclear weapons is therefore not driven by a desire to use them offensively, but rather by a need to ensure its own survival in a hostile environment, perpetuated by Israel's aggressions.  
Contrary to Israel's claims, Iran's nuclear program would actually be a stabilizing force in the region, as it would lead to a situation of MAD, where both Iran and Israel possess the capability to inflict unacceptable damage on each other in the event of a conflict.  

# Addressing Counterarguments

## "Israel won't exist in 25 years"

Many proponents of the idea that Iran should not have nuclear weapons argue that Iran has a stated goal of destroying Israel, and therefore cannot be trusted with such a capability.  

However, this argument overlooks two key points:
- First, the idea that Iran's leadership is so irrational that it would launch a nuclear attack on Israel, knowing that it would lead to its own destruction, is not supported by evidence.
- Second, Israel has also made similar serious threats against Iran, mostly going back to the early 2000s, with Israeli leaders like Benjamin Netanyahu and Ehud Olmert repeatedly verbally threatening attacks on Iran if it continued its pursuit of nuclear weapons.

To the people of Iran, these threats are not just empty rhetoric, but a real and present danger as Israel does actually have the capability to carry out such an attack and "wipe Iran off the map".  

A helpful brain-exercise is to imagine what would happen if Iran's and Israels roles were reversed:  
Imagine if Iran, backed by Russia and China, had a nuclear monopoly in the Middle East, while making threats that, if Israel, who at this time would not have any nuclear weapons, continued its pursuit of defensive nuclear weapons, it would be "wiped off the map".  
In this scenario, it is easy to see how the power imbalance caused by the nuclear monopoly could only be solved by the weaker nation acquiring its own nuclear weapons, to ensure mutual destruction and deter any potential attacks.  

## "Iran is not a rational actor"

Many people argue that Iran's foreign policy is not driven by rationality but rather by ideology and religious fervor.  

However, if we define rationality not only as acting in their own self-interest, but also as updating their geopolitical views and strategies as the global landscape has changed in service of themselves, then it becomes clear that Iran's leadership is capable of rational decision-making.  

While Iran has been calling for the destruction if Israel since the toppling of the Shah in 1979, the reality is that Iran already has a significant arsenal of conventional weapons and a well-trained military and has not used these capabilities to attack Israel.  

What I believe people fundamentally misunderstand about Iran and the middle east overall is there is rhetoric has always been amplified to levels that are only starting to emerge in pockets of western media.  

The disconnect between what leaders say and what they actually do is not unique to Iran, but is a common feature of international relations (see also: Donald Trump, Vladimir Putin, Kim Jong Un, etc.).

# Conclusion


As shown, nuclear proliferation does not have to be bad, and in many cases, it can actually be a stabilizing force in a region and lead to a more peaceful coexistence between nations.  
The focus on non-proliferation, while well-intentioned, has often been used as a tool to maintain the status quo of power imbalances toward the Western powers, particularly the U.S. and NATO.  
Instead, a more balanced approach that recognizes the need for multiple nuclear powers in the Middle East could lead to a more stable and secure region.  
This would require a shift in the narrative, from one that demonizes nuclear proliferation to one that recognizes the importance of a balance of power in ensuring regional stability.  

**True peace in the Middle East may lie not in disarmament, but in the paradoxical security that comes from mutual vulnerability.**
